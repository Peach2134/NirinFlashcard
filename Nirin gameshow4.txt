let allCards = DATA, cards = DATA, current = 0, flipped = false;

function filterCategory(cat){
  const buttons = document.querySelectorAll('#left-sidebar h3+button, #left-sidebar h3+button+button, #left-sidebar h3+button+button+button, #left-sidebar h3+button+button+button+button');
  buttons.forEach(b=>b.classList.remove('active'));
  event.target.classList.add('active');
  if(cat === 'all') cards = allCards;
  else cards = allCards.filter(c => c.category === cat);
  current = 0;
  renderCard();
  renderCardList();
}
function renderCard(){
  const cardEl = document.getElementById('card');
  cardEl.classList.remove('flipped');
  flipped = false;
  document.getElementById('front').innerHTML = cards[current].front;
  document.getElementById('back').innerHTML = '?';
}

function flipCard(){
  const cardEl = document.getElementById('card');
  if(!flipped){
    document.getElementById('back').innerHTML = cards[current].back;
    cardEl.classList.add('flipped'); flipped=true;
  } else {
    cardEl.classList.remove('flipped'); flipped=false;
  }
}

function nextCard(){ current=(current+1)%cards.length; renderCard(); }
function prevCard(){ current=(current-1+cards.length)%cards.length; renderCard(); }
function shuffleCards(){ cards=cards.sort(()=>Math.random()-0.5); current=0; renderCard(); renderCardList(); }
function speakCard(){ 
  const u=new SpeechSynthesisUtterance(cards[current].front.replace(/<br>/g,' ')); 
  u.lang='en-US'; 
  speechSynthesis.speak(u); 
}

function renderCardList(){
  const list=document.getElementById('cardList'); list.innerHTML='';
  cards.forEach((c,i)=>{
    const li=document.createElement('li');
    li.innerHTML=c.front;
    li.onclick=()=>{current=i; renderCard();};
    list.appendChild(li);
  });
}

function toggleRight(){ document.getElementById('right-sidebar').classList.toggle('hidden'); }

}

function filterLevel(level){
  const buttons = document.querySelectorAll('#left-sidebar button');
  buttons.forEach(b=>b.classList.remove('active'));
  event.target.classList.add('active');
  
  if(level === 'all') cards = allCards;
  else cards = allCards.filter(c => c.level === level);
  
  current = 0;
  renderCard();
  renderCardList();
}

document.addEventListener('keydown', e=>{
  if(e.code==='Space'){ e.preventDefault(); flipCard(); }
  if(e.code==='ArrowRight') nextCard();
  if(e.code==='ArrowLeft') prevCard();
  if(e.key.toLowerCase()==='s') shuffleCards();
  if(e.key.toLowerCase()==='p') speakCard();
});
